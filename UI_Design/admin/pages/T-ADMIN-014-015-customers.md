## Data Requirements
```typescript
interface CustomerListData {
  customers: CustomerListItem[];
  filters: {
    segment: CustomerSegment | 'all';
    location: string;
    ordersRange: [number, number];
    spentRange: [number, number];
    search: string;
  };
  pagination: PaginationInfo;
  stats: {
    total: number;
    new: number;
    returning: number;
    vip: number;
  };
}

interface CustomerDetailData {
  customer: Customer;
  stats: {
    totalSpent: number;
    totalOrders: number;
    avgOrderValue: number;
  };
  orders: Order[];
  addresses: Address[];
  notes: CustomerNote[];
  tags: string[];
  purchaseHistory: MonthlyPurchase[];
}
```
